function setupSmoothVerticalSlider(sliderId, speed = 0.5) {
    const slider = document.getElementById(sliderId);
    const parent = slider.parentElement;
    const imgs = Array.from(slider.querySelectorAll('img'));
    let loadedCount = 0;
    imgs.forEach(img => { img.onload = img.onerror = () => { loadedCount++; if(loadedCount===imgs.length)initSlider(); }; });

    function initSlider() {
        const totalImgs = imgs.length;
        const imgHeight = imgs[0].offsetHeight;
        imgs.forEach(img => slider.appendChild(img.cloneNode(true)));
        let scrollTop = 0, isPaused = false;
        parent.addEventListener('mouseenter',()=>isPaused=true);
        parent.addEventListener('mouseleave',()=>isPaused=false);
        function animate() {
            if(!isPaused){ scrollTop += speed; if(scrollTop >= imgHeight * totalImgs) scrollTop=0; slider.style.transform=`translateY(-${scrollTop}px)`; }
            requestAnimationFrame(animate);
        }
        requestAnimationFrame(animate);
        window.addEventListener('resize',()=>{ slider.style.transition='none'; scrollTop=0; });
    }
}
setupSmoothVerticalSlider('slider1',0.5);
setupSmoothVerticalSlider('slider2',0.5);
setupSmoothVerticalSlider('slider3',0.5);
